package com.example.poweraverageapp;

import androidx.appcompat.app.AppCompatActivity;
import android.content.Intent;
import android.os.Bundle;
import android.widget.*;

public class MainActivity extends AppCompatActivity {

    EditText etNumber1, etNumber2;
    Button btnPower, btnAverage;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        etNumber1 = findViewById(R.id.etNumber1);
        etNumber2 = findViewById(R.id.etNumber2);
        btnPower = findViewById(R.id.btnPower);
        btnAverage = findViewById(R.id.btnAverage);

        btnPower.setOnClickListener(v -> handleCalculation("power"));

        btnAverage.setOnClickListener(v -> handleCalculation("average"));
    }

    private void handleCalculation(String operation) {
        String n1 = etNumber1.getText().toString().trim();
        String n2 = etNumber2.getText().toString().trim();

        if (n1.isEmpty() || n2.isEmpty()) {
            Toast.makeText(this, "Please enter both numbers", Toast.LENGTH_SHORT).show();
            return;
        }

        double num1 = Double.parseDouble(n1);
        double num2 = Double.parseDouble(n2);

        double result;

        if (operation.equals("power")) {
            result = Math.pow(num1, num2);
        } else { // average
            result = (num1 + num2) / 2;
        }

        // Send result to ResultActivity
        Intent intent = new Intent(MainActivity.this, ResultActivity.class);
        intent.putExtra("result", result);
        intent.putExtra("operation", operation);
        startActivity(intent);
    }
}

package com.example.radiobuttondemo;

import android.os.Bundle;
import android.widget.*;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    RadioGroup radioGroupClass;
    RadioButton rbFY, rbSY, rbTY;
    CheckBox cbCpp, cbDS, cbDotNet, cbJava;
    Button btnDisplay, btnClear;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        // Binding Views
        radioGroupClass = findViewById(R.id.radioGroupClass);
        rbFY = findViewById(R.id.rbFY);
        rbSY = findViewById(R.id.rbSY);
        rbTY = findViewById(R.id.rbTY);

        cbCpp = findViewById(R.id.cbCpp);
        cbDS = findViewById(R.id.cbDS);
        cbDotNet = findViewById(R.id.cbDotNet);
        cbJava = findViewById(R.id.cbJava);

        btnDisplay = findViewById(R.id.btnDisplay);
        btnClear = findViewById(R.id.btnClear);

        // Display Button Click
        btnDisplay.setOnClickListener(v -> {
            StringBuilder result = new StringBuilder();

            // Get selected RadioButton
            int selectedId = radioGroupClass.getCheckedRadioButtonId();
            if (selectedId != -1) {
                RadioButton selectedRadio = findViewById(selectedId);
                result.append(selectedRadio.getText().toString()).append("\n");
            } else {
                result.append("No class selected\n");
            }

            // Add selected CheckBoxes
            if (cbCpp.isChecked()) result.append("C++\n");
            if (cbDS.isChecked()) result.append("DS USING C\n");
            if (cbDotNet.isChecked()) result.append("C#.NET\n");
            if (cbJava.isChecked()) result.append("JAVA\n");

            // Show result
            Toast.makeText(MainActivity.this, result.toString(), Toast.LENGTH_LONG).show();
        });

        // Clear Button Click
        btnClear.setOnClickListener(v -> {
            radioGroupClass.clearCheck();
            cbCpp.setChecked(false);
            cbDS.setChecked(false);
            cbDotNet.setChecked(false);
            cbJava.setChecked(false);
        });
    }
}

package com.example.stringoperationapp;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.widget.*;

public class MainActivity extends AppCompatActivity {

    EditText etInput, etOutput;
    RadioGroup radioGroup;
    RadioButton radioUpper, radioLower, radioRight, radioLeft;
    Button btnClick;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        
        // Bind views
        etInput = findViewById(R.id.etInput);
        etOutput = findViewById(R.id.etOutput);
        radioGroup = findViewById(R.id.radioGroup);
        radioUpper = findViewById(R.id.radioUpper);
        radioLower = findViewById(R.id.radioLower);
        radioRight = findViewById(R.id.radioRight);
        radioLeft = findViewById(R.id.radioLeft);
        btnClick = findViewById(R.id.btnClick);

        btnClick.setOnClickListener(view -> {
            String inputText = etInput.getText().toString();
            String result = "";

            if (inputText.isEmpty()) {
                Toast.makeText(MainActivity.this, "Please enter a string", Toast.LENGTH_SHORT).show();
                return;
            }

            // Identify selected radio button
            int selectedId = radioGroup.getCheckedRadioButtonId();

            if (selectedId == R.id.radioUpper) {
                result = inputText.toUpperCase();
            } else if (selectedId == R.id.radioLower) {
                result = inputText.toLowerCase();
            } else if (selectedId == R.id.radioRight) {
                if (inputText.length() > 5) {
                    result = inputText.substring(inputText.length() - 5);
                } else {
                    result = inputText;
                }
            } else if (selectedId == R.id.radioLeft) {
                if (inputText.length() > 5) {
                    result = inputText.substring(0, 5);
                } else {
                    result = inputText;
                }
            }

            // Set and show output
            etOutput.setText(result);
            Toast.makeText(MainActivity.this, "Result: " + result, Toast.LENGTH_SHORT).show(); 
        });
    }
}

public class MainActivity extends AppCompatActivity {

    EditText etInput;
    Button btnShowImage;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        
        etInput = findViewById(R.id.etInput);
        btnShowImage = findViewById(R.id.btnShowImage);

        btnShowImage.setOnClickListener(v -> {
            String input = etInput.getText().toString().trim();
            if (input.isEmpty()) {
                Toast.makeText(this, "Please enter a number 1-4", Toast.LENGTH_SHORT).show();
                return;
            }

            int selected = Integer.parseInt(input);
            if (selected < 1 || selected > 4) {
                Toast.makeText(this, "Invalid number! Choose 1 to 4", Toast.LENGTH_SHORT).show();
                return;
            }

            // Invoke ImageActivity
            Intent intent = new Intent(MainActivity.this, ImageActivity.class);
            intent.putExtra("image_id", selected);
            startActivity(intent);
        });
    }
}

package com.example.imageswitcherdemo;

import android.os.Bundle;
import android.view.View;
import android.view.animation.AnimationUtils;
import android.widget.Button;
import android.widget.ImageSwitcher;
import android.widget.ImageView;
import android.widget.ViewSwitcher;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    ImageSwitcher imageSwitcher;
    Button btnPrev, btnNext;

    int imageIds[] = {
            R.drawable.img1,
            R.drawable.img2,
            R.drawable.img3
    };

    int currentIndex = 0;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        
        imageSwitcher = findViewById(R.id.imageSwitcher);
        btnPrev = findViewById(R.id.btnPrev);
        btnNext = findViewById(R.id.btnNext);

        // Set factory for ImageSwitcher
        imageSwitcher.setFactory(() -> {
            ImageView imageView = new ImageView(getApplicationContext());
            imageView.setScaleType(ImageView.ScaleType.FIT_CENTER);
            imageView.setLayoutParams(new 
                ImageSwitcher.LayoutParams(
                    ImageSwitcher.LayoutParams.MATCH_PARENT,
                    ImageSwitcher.LayoutParams.MATCH_PARENT));
            return imageView;
        });

        // Set animations (optional)
        imageSwitcher.setInAnimation(AnimationUtils.loadAnimation(this, android.R.anim.slide_in_left));
        imageSwitcher.setOutAnimation(AnimationUtils.loadAnimation(this, android.R.anim.slide_out_right));

        // Set initial image
        imageSwitcher.setImageResource(imageIds[currentIndex]);

        // Prev Button
        btnPrev.setOnClickListener(view -> {
            currentIndex--;
            if (currentIndex < 0) {
                currentIndex = imageIds.length - 1;
            }
            imageSwitcher.setImageResource(imageIds[currentIndex]);
        });

        // Next Button
        btnNext.setOnClickListener(view -> {
            currentIndex++;
            if (currentIndex >= imageIds.length) {
                currentIndex = 0;
            }
            imageSwitcher.setImageResource(imageIds[currentIndex]);
        });
    }
}

package com.example.decimaltobinaryconverter;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    TextView tvDecimal, tvBinary;
    StringBuilder input = new StringBuilder();  // to store user input

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        tvDecimal = findViewById(R.id.tvDecimal);
        tvBinary = findViewById(R.id.tvBinary);

        // Button IDs
        int[] btnIds = {
                R.id.btn0, R.id.btn1, R.id.btn2, R.id.btn3, R.id.btn4,
                R.id.btn5, R.id.btn6, R.id.btn7, R.id.btn8, R.id.btn9
        };

        View.OnClickListener numClickListener = view -> {
            Button b = (Button) view;
            input.append(b.getText().toString());
            updateConversion();
        };

        for (int id : btnIds) {
            findViewById(id).setOnClickListener(numClickListener);
        }

        // Clear Button
        findViewById(R.id.btnAC).setOnClickListener(v -> {
            input.setLength(0);
            updateConversion();
        });

        // Delete Button
        findViewById(R.id.btnDel).setOnClickListener(v -> {
            if (input.length() > 0) {
                input.deleteCharAt(input.length() - 1);
                updateConversion();
            }
        });

        // Initial update
        updateConversion();
    }

    private void updateConversion() {
        String decimalStr = input.toString();
        tvDecimal.setText(decimalStr.isEmpty() ? "0" : decimalStr);

        try {
            int decimalValue = Integer.parseInt(decimalStr);
            tvBinary.setText(Integer.toBinaryString(decimalValue));
        } catch (NumberFormatException e) {
            tvBinary.setText("0");
        }
    }
}
